Build the SPDE mesh on UTM coordinates

Code 
# Coordinates matrix for mesh
coords <- cbind(spatial_binom$utm_x, spatial_binom$utm_y)

# Create a 2D mesh (tune max.edge / cutoff as needed)
mesh <- inla.mesh.2d(
  loc = coords,
  max.edge = c(100000, 300000),   # in metres (since UTM)
  cutoff  = 20000,                # min distance between points
  offset  = c(50000, 200000)      # inner/outer extension of mesh
)

plot(mesh)
points(coords, col = "red", pch = 16, cex = 0.6)

Convert data to sf object

Code 
spatial_binom_sf <- spatial_binom %>% st_as_sf(coords = c("utm_x", "utm_y"), crs =  32632)

Define the SPDE model (PC prior)

Code 
spde <- inla.spde2.pcmatern(
  mesh = mesh,
  prior.range = c(300000, 0.5),  # P(range < 300 km) = 0.5
  prior.sigma = c(1.0, 0.01)     # P(sigma > 1) = 0.01
)

Define the inlabru components

We model:

  • Fixed effects for elevation, ndvi, urban

  • A spatial random field field(geometry, model = spde)

  • An intercept

Code 
components <- ~
  Intercept(1, model = "linear") +
  beta_elev(elevation, model = "linear") +
  beta_ndvi(ndvi, model = "linear") +
  beta_urban(urban, model = "linear") +
  field(geometry, model = spde)

lik <- like(family = "binomial", data = spatial_binom_sf, Ntrials = n_tested,
            formula = n_pos  ~
  Intercept +
  beta_elev +
  beta_ndvi +
  beta_urban +
  field)

Fit the spatial binomial model

Response = n_pos, trials = n_tested, family = “binomial”.

Code 
fit_spatial_binom <- bru(
  components = components,
  lik,
  options    = list(
    control.compute = list(dic = TRUE, waic = TRUE, cpo = TRUE)
  )
)

Make predictions on a grid (prevalence surface)

We start by loading the prediction grid with UTM coords and covariates. Also, convert it to sf object.

Code 
pred_grid <- read_csv("data/prediction_grid_utm.csv") %>% 
  st_as_sf(coords = c("utm_x", "utm_y"), crs =  32632)
Code 
# Logistic helper
logistic <- function(x) 1 / (1 + exp(-x))

# Prediction: posterior mean of prevalence
pred_res <- predict(
  fit_spatial_binom,
  newdata = pred_grid,
  formula = ~ logistic(
    Intercept +
      beta_elev * elevation +
      beta_ndvi * ndvi +
      beta_urban * urban +
      field
  ),
  n.samples = 1000   # increase for smoother estimates
)

# Combine predictions with grid
names(pred_res)[names(pred_res) == "mean"] <- "prev_mean"
names(pred_res)[names(pred_res) == "sd"]   <- "prev_sd"
