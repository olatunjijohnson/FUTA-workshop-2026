% Optional: make links purple too
\hypersetup{
  colorlinks=true,
  linkcolor=uomPurple,
  urlcolor=uomPurple,
  citecolor=uomPurple
}

    \begin{beamercolorbox}[wd=.2\paperwidth,ht=2.5ex,dp=1.0ex,right]{date in head/foot}%
      \insertframenumber{} / \inserttotalframenumber\hspace{1em}
    \end{beamercolorbox}%
